version (versão do git)

add (adiciona ao index)
status (status do git)
commit (comita)
log (--oneline --decorate -- all --graph)
branch (cria novo branch - branch <nome do branch> - branch -d <nome do branch> (apaga))
checkout (muda o branch - checkout <nome do branch>
cat (mostra conteudo do arquivo)
cat ./git/HEADE (mostra qual branch atual)
echo e > e.txt (cria um arquivo e.txt com um conteudo "e")
merge (se você estiver no master - git merge <nome da branch>)
ls -la (lista os arquivos do branch atual)
stach (save:criar; apply:tras de volta; list:lista;drop:apagar;pop:desenpilhar (apply + drop))
